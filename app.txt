    <script>
        const MENU = [
            { id: 1, name: "Carne", price: 9, category: "Espetos" },
            { id: 2, name: "Carne c/ Bacon", price: 10, category: "Espetos" },
            { id: 3, name: "Kafta", price: 8, category: "Espetos" },
            { id: 4, name: "Kafta c/ Queijo", price: 10, category: "Espetos" },
            { id: 5, name: "Frango c/ Bacon", price: 9, category: "Espetos" },
            { id: 6, name: "Coração", price: 10, category: "Espetos" },
            { id: 7, name: "Linguiça", price: 8, category: "Espetos" },
            { id: 8, name: "Linguiça Apimentada", price: 8, category: "Espetos" },
            { id: 9, name: "Queijo Coalho", price: 9, category: "Espetos" },
            { id: 10, name: "Pão de Alho", price: 6, category: "Espetos" },
            { id: 50, name: "Opcional: No Lanche", price: 6, category: "Adicionais" },
            { id: 100, name: "Refri Lata", price: 6, category: "Bebidas" },
            { id: 101, name: "Suco Del Valle", price: 8, category: "Bebidas" },
            { id: 102, name: "Heineken/Original", price: 6, category: "Bebidas" },
            { id: 103, name: "Amstel", price: 5, category: "Bebidas" },
            { id: 104, name: "Skol", price: 4, category: "Bebidas" },
            { id: 105, name: "Gin (700ml)", price: 15, category: "Bebidas" },
            { id: 106, name: "Whisky (700ml)", price: 20, category: "Bebidas" },
            { id: 107, name: "Mansão Maromba", price: 25, category: "Bebidas" },
            { id: 108, name: "Heineken - Long Neck", price: 10, category: "Bebidas" }
        ];

        let state = { activeOrders: [], salesHistory: [], currentView: 'orders', selectedOrderId: null };

        function init() {
            const saved = localStorage.getItem('espetinho_pdv_state');
            if (saved) state = JSON.parse(saved);
            renderOrders();
            renderReport();
        }

        function save() { localStorage.setItem('espetinho_pdv_state', JSON.stringify(state)); }

        function switchView(view) {
            document.querySelectorAll('section').forEach(s => s.classList.add('hidden'));
            document.getElementById(`view-${view}`).classList.remove('hidden');
            document.getElementById('bottom-bar').classList.toggle('hidden', view !== 'menu');
            document.getElementById('tab-orders').classList.toggle('active-tab', view === 'orders');
            document.getElementById('tab-report').classList.toggle('active-tab', view === 'report');
            if(view === 'orders') renderOrders();
            if(view === 'report') renderReport();
        }

        function openModal(id) { document.getElementById(id).classList.remove('hidden'); }
        function closeModals() { document.querySelectorAll('[id^="modal-"]').forEach(m => m.classList.add('hidden')); }

        function createNewOrder() {
            const name = document.getElementById('input-customer-name').value.trim().toUpperCase();
            if (!name) return;
            const order = { id: Date.now(), customer: name, items: {} };
            state.activeOrders.push(order);
            save();
            document.getElementById('input-customer-name').value = '';
            closeModals();
            openOrderDetails(order.id);
        }

        function openOrderDetails(id) {
            state.selectedOrderId = id;
            const order = state.activeOrders.find(o => o.id === id);
            document.getElementById('menu-customer-name').innerText = order.customer;
            renderMenu();
            switchView('menu');
        }

        function renderOrders() {
            const list = document.getElementById('orders-list');
            if (!state.activeOrders.length) { 
                list.innerHTML = '<div class="text-center py-20 text-slate-700 italic text-sm">Nenhuma comanda aberta.</div>'; 
                return; 
            }
            list.innerHTML = state.activeOrders.map(o => `
                <div onclick="openOrderDetails(${o.id})" class="bg-slate-900 p-4 rounded-xl border border-slate-800 flex justify-between items-center active:bg-slate-800 cursor-pointer">
                    <span class="font-bold text-slate-200">${o.customer}</span>
                    <span class="text-orange-500 font-black">R$ ${calcTotal(o).toFixed(2)}</span>
                </div>
            `).join('');
        }

        function renderMenu() {
            const container = document.getElementById('menu-categories');
            const order = state.activeOrders.find(o => o.id === state.selectedOrderId);
            const cats = [...new Set(MENU.map(i => i.category))];
            
            container.innerHTML = cats.map(cat => `
                <h3 class="text-[10px] text-slate-500 font-bold uppercase tracking-widest mb-2">${cat}</h3>
                <div class="grid gap-2 mb-4">
                    ${MENU.filter(i => i.category === cat).map(item => `
                        <div class="bg-slate-900 p-3 rounded-xl border border-slate-800 flex justify-between items-center">
                            <div class="flex-1 cursor-pointer" onclick="updateQty(${item.id}, 1)">
                                <p class="font-bold text-sm text-slate-200">${item.name}</p>
                                <p class="text-orange-500 font-bold text-xs">R$ ${item.price.toFixed(2)}</p>
                            </div>
                            <div class="flex items-center gap-3">
                                ${order.items[item.id] ? `
                                    <button onclick="updateQty(${item.id}, -1)" class="w-10 h-10 bg-slate-800 rounded-lg text-slate-400 font-bold active:bg-slate-700">-</button>
                                    <span class="font-black text-white w-4 text-center">${order.items[item.id]}</span>
                                ` : ''}
                                <button onclick="updateQty(${item.id}, 1)" class="w-12 h-12 bg-orange-600 rounded-lg font-black text-xl active:bg-orange-500">+</button>
                            </div>
                        </div>
                    `).join('')}
                </div>
            `).join('');
            updateTotalBar();
        }

        function updateQty(itemId, change) {
            const order = state.activeOrders.find(o => o.id === state.selectedOrderId);
            order.items[itemId] = (order.items[itemId] || 0) + change;
            if (order.items[itemId] <= 0) delete order.items[itemId];
            save();
            renderMenu();
        }

        function calcTotal(order) {
            return Object.entries(order.items).reduce((acc, [id, qty]) => {
                const item = MENU.find(i => i.id == id);
                return acc + (item.price * qty);
            }, 0);
        }

        function updateTotalBar() {
            const order = state.activeOrders.find(o => o.id === state.selectedOrderId);
            document.getElementById('order-total-display').innerText = `R$ ${calcTotal(order).toFixed(2)}`;
            document.getElementById('checkout-total').innerText = `R$ ${calcTotal(order).toFixed(2)}`;
        }

        function finishOrder(method) {
            const idx = state.activeOrders.findIndex(o => o.id === state.selectedOrderId);
            const order = state.activeOrders[idx];
            
            state.salesHistory.push({
                total: calcTotal(order),
                method,
                items: JSON.parse(JSON.stringify(order.items)),
                date: new Date().toLocaleTimeString()
            });

            state.activeOrders.splice(idx, 1);
            save();
            closeModals();
            switchView('orders');
        }

        function renderReport() {
            const summary = state.salesHistory.reduce((acc, s) => {
                acc[s.method] += s.total;
                acc.total += s.total;
                Object.entries(s.items).forEach(([id, qty]) => {
                    acc.itemCounts[id] = (acc.itemCounts[id] || 0) + qty;
                });
                return acc;
            }, { PIX: 0, Cartão: 0, Dinheiro: 0, total: 0, itemCounts: {} });

            document.getElementById('rep-total').innerText = `R$ ${summary.total.toFixed(2)}`;
            document.getElementById('rep-count').innerText = state.salesHistory.length;
            document.getElementById('rep-pix').innerText = `R$ ${summary.PIX.toFixed(2)}`;
            document.getElementById('rep-card').innerText = `R$ ${summary.Cartão.toFixed(2)}`;
            document.getElementById('rep-cash').innerText = `R$ ${summary.Dinheiro.toFixed(2)}`;

            const listContainer = document.getElementById('rep-items-list');
            const sortedItems = Object.entries(summary.itemCounts).sort((a, b) => b[1] - a[1]);
            
            if (sortedItems.length === 0) {
                listContainer.innerHTML = '<div class="p-4 text-slate-600 italic text-xs">Sem itens vendidos ainda.</div>';
            } else {
                listContainer.innerHTML = sortedItems.map(([id, qty]) => {
                    const item = MENU.find(i => i.id == id);
                    return `<div class="p-4 flex justify-between"><span>${item.name}</span><span class="font-black text-orange-400">${qty} un</span></div>`;
                }).join('');
            }
        }

        async function exportAndReset() {
            closeModals(); // Fecha modal de confirmação
            if (!state.salesHistory.length) return alert("Sem dados para exportar.");
            
            const summary = state.salesHistory.reduce((acc, s) => {
                acc[s.method] += s.total;
                acc.total += s.total;
                Object.entries(s.items).forEach(([id, qty]) => {
                    acc.itemCounts[id] = (acc.itemCounts[id] || 0) + qty;
                });
                return acc;
            }, { PIX: 0, Cartão: 0, Dinheiro: 0, total: 0, itemCounts: {} });

            const itemText = Object.entries(summary.itemCounts)
                .sort((a, b) => b[1] - a[1])
                .map(([id, qty]) => `• ${qty}x ${MENU.find(i => i.id == id).name}`)
                .join('\n');

            const texto = `🔥 *FECHAMENTO ESPETINHO VIP*\n📅 ${new Date().toLocaleDateString()}\n\n💰 *Total: R$ ${summary.total.toFixed(2)}*\n------------------\n💎 PIX: R$ ${summary.PIX.toFixed(2)}\n💳 Cartão: R$ ${summary.Cartão.toFixed(2)}\n💵 Dinheiro: R$ ${summary.Dinheiro.toFixed(2)}\n\n📊 *ITENS VENDIDOS:*\n${itemText}\n------------------\n✅ Total de Vendas: ${state.salesHistory.length}`;

            try {
                if (navigator.share) {
                    await navigator.share({ text: texto });
                    finalReset();
                } else { throw "No Share API"; }
            } catch (err) {
                // Fallback WhatsApp Link
                window.open(`https://api.whatsapp.com/send?text=${encodeURIComponent(texto)}`, '_blank');
                
                // Copy to clipboard
                const el = document.createElement('textarea');
                el.value = texto;
                document.body.appendChild(el);
                el.select();
                document.execCommand('copy');
                document.body.removeChild(el);

                showToast();
                setTimeout(() => {
                    finalReset();
                }, 1500);
            }
        }

        function finalReset() {
            state.salesHistory = [];
            save();
            renderReport();
            switchView('orders');
        }

        function showToast() {
            const t = document.getElementById('toast');
            t.classList.remove('hidden');
            setTimeout(() => t.classList.add('hidden'), 3000);
        }

        window.onload = init;
    </script>
